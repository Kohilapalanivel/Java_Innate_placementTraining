1. Write a Java program to calculate the electricity bill based on the following rules:
Units	Rate per unit
0 – 100	₹5
101 – 300	₹7
Above 300	₹10
If total bill > ₹2000, add 10% surcharge.


import java.util.Scanner;
class Main {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int unit = s.nextInt();
        double price = 0;
        double  total_bill = 0;
        if(unit <= 100)
        {
            price = unit*5;
        }
        else if(unit > 100 && unit<=300)
        {
            price = (100*5) + ((unit - 100)*7);
        }
        else
        {
            price = (100*5) + (200*7) + ((unit - 300)*10);
        }
        if (price > 2000)
        {
            double charge = 0.20 * price;
            total_bill = price + charge;
        }
        else
        {
            total_bill = price;
        }
        s.close();
        System.out.println("The Electricity bill is: " + total_bill);
    }
}


2.  Student Grade System
Get marks for 5 subjects.
Find average and assign grade as per the following:
Average	Grade
≥ 90	A+
80–89	A
70–79	B
60–69	C
< 60	Fail


import java.util.Scanner;
class Main {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int m1 = s.nextInt();
        int m2 = s.nextInt();
        int m3 = s.nextInt();
        int m4 = s.nextInt();
        int m5 = s.nextInt();
        int sum = m1+m2+m3+m4+m5;
        int avg = sum / 5;
        if(avg>=90)
        {
            System.out.println("A+");
        }
        else if(avg<=89 && avg>=80)
        {
            System.out.println("A");
        }
        else if(avg<=79 && avg >=70)
        {
            System.out.println("B");
        }
        else if(avg<=69 && avg>=60)
        {
            System.out.println("C");
        }
        else
        {
            System.out.println("Fail");
        }
        s.close();
    }
}

3. Employee Bonus Calculation
Company policy: 
If experience > 5 years → 15% bonus
Else if experience ≥ 3 → 10% bonus
Else → 5% bonus

class Main {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int exp = s.nextInt();
        double salary = s.nextDouble();
        double bonus;
        double total_salary;
        if(exp > 5)
        {
            bonus = 0.15*salary;
            total_salary = salary + bonus;
        }
        else if(exp >=3)
        {
            bonus = 0.10*salary;
            total_salary = salary + bonus;
        }
        else
        {
            bonus = 0.05*salary;
            total_salary = salary + bonus;
        }
        System.out.println("Salary is: "+ total_salary);
        s.close();
    }
}

5. Shopping Discount System
If total amount > ₹5000, apply 20% discount,
else if > ₹2000 → 10% discount,
otherwise no discount.

import java.util.Scanner;
class Main {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int bill = s.nextInt();
        double total_bill;
        double discount;
        if(bill > 5000)
        {
            discount = 0.20*bill;
            total_bill = bill - discount;
        }
        else if(bill>2000)
        {
            discount = 0.10*bill;
            total_bill = bill - discount;
        }
        else
        {
            total_bill = bill;
        }
        System.out.println("The total Bill  after Discount is: " + total_bill);
        s.close();
    }
}

7.Simple Calculator using Switch

import java.util.Scanner;
public class Main
{
	public static void main(String[] args) {
	    Scanner s = new Scanner(System.in);
	    int n1 = s.nextInt();
	    int n2 = s.nextInt();
	    char cal = s.next().charAt(0);
	    switch(cal)
	    {
	        case '+':
	            System.out.println("The Addition of two numbers = " + (n1+n2));
	            break;
	        case '-':
	                System.out.println("The Difference of two numbers = " +(n1-n2));
	                break;
	        case '*':
	                    System.out.println("The Multiplication of two numbers = " +(n1*n2));
	                    break;
	        case '/':
	                    System.out.println("The Division of two numbers = " +(n1/n2));
	                    break;
	        case '%':
	                    System.out.println("The Modulo of two numbers = " +(n1% n2));
	                    break;
	         default:
	               System.out.println("Invalid Operator! Please Enter valid operator");
	               break;
	                    
	            
	    }
	}
}
