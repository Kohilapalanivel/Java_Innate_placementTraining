1.Print even number from 5 to 55

import java.util.Scanner;
public class Main
{
	public static void main(String[] args) {
	    System.out.println("The Even Numbers from 5 to 55:");
	    for(int i=5;i<=55;i++)
	    {
	        if(i%2 == 0)
	        {
	            System.out.print(i + " ");
	        }
	    }
	}
}


2.multiplication table from 1 to 20(for user entered number)

import java.util.Scanner;
public class Main
{
	public static void main(String[] args) {
	    Scanner s = new Scanner(System.in);
	    int n = s.nextInt();
	    System.out.println("The Multiplication table of number " + n+ " is:");
	    for(int i=1;i<=20;i++)
	    {
	        System.out.println(i + " x " + n + " = " + (i*n));
	    }
	}
}


3.password retry system(using while loop+condition)

import java.util.Scanner;
public class Main
{
	public static void main(String[] args) {
	    Scanner s = new Scanner(System.in);
	    int attempt = 0;
	    while(attempt < 3)
	    {
	        String pass = s.nextLine();
	        if(pass.equals("456523kohi"))
	   
	        {
	            System.out.println("Access Granted! You have logged in Successfully");
	            break;
	        }
	        else
	        {
	            attempt++;
	            if(attempt<3)
	            {
	            System.out.println("Incorrect Password! Attempts left: " + (3-attempt));
	            }
	            else{
	                System.out.println("Too many Failed attempts..Acces Denied!");
	            }
	        }
	        
	    }
	}
}



4.factorial of a number


import java.util.Scanner;
public class Main
{
	public static void main(String[] args) {
	    Scanner s = new Scanner(System.in);
	    int n = s.nextInt();
	    int fact =1;
	    for(int i=1;i<=n;i++)
	    {
	        fact*=i;
	    }
	    System.out.println("The factorial of the number "+ n + " is " + fact); 
	}
}



5.print all prime  number between 1 and 10

public class Main {
    public static void main(String[] args) 
    {
        for (int num = 2; num <= 10; num++) 
        { 
            boolean isPrime = true;

            for (int i = 2; i <= Math.sqrt(num); i++)
            {
                if (num % i == 0)
                {
                    isPrime = false;
                    break;
                }
            }

            if (isPrime)
            {
                System.out.println(num);
            }
        }
    }
}
